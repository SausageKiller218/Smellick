BEGIN -- Slonyari
USE pubs;
DECLARE @max INT SELECT @max = MAX(i)+1 FROM nums 
DECLARE @end INT SET @end = 2*@max-2
While @max <= @end
    BEGIN
    INSERT INTO nums(i) VALUES(@max)
    SET @max = @max + 1
    END
END
